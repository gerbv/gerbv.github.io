55 gerbv_render_info_t gerbv_export_autoscale_project (gerbv_project_t *gerbvProject) {
56  gerbv_render_size_t bb;
57  gerbv_render_get_boundingbox(gerbvProject, &bb);
58  // add a border around the bounding box
59  gfloat tempWidth = bb.right - bb.left;
60  gfloat tempHeight = bb.bottom - bb.top;
61  bb.right += (tempWidth*0.05);
62  bb.left -= (tempWidth*0.05);
63  bb.bottom += (tempHeight*0.05);
64  bb.top -= (tempHeight*0.05);
65  float width = bb.right - bb.left + 0.001; // Plus a little extra to prevent from
66  float height = bb.bottom - bb.top + 0.001; // missing items due to round-off errors
67 
68  gerbv_render_info_t renderInfo = {72, 72, bb.left, bb.top,
69  GERBV_RENDER_TYPE_CAIRO_HIGH_QUALITY, width*72, height*72};
70  return renderInfo;
71 }

83 void gerbv_export_png_file_from_project (gerbv_project_t *gerbvProject, gerbv_render_info_t *renderInfo, gchar const* filename) {
84  cairo_surface_t *cSurface = cairo_image_surface_create (CAIRO_FORMAT_ARGB32,
85  renderInfo->displayWidth, renderInfo->displayHeight);
86  cairo_t *cairoTarget = cairo_create (cSurface);
87  gerbv_render_all_layers_to_cairo_target (gerbvProject, cairoTarget, renderInfo);
88  if (CAIRO_STATUS_SUCCESS != cairo_surface_write_to_png (cSurface, filename)) {
89  GERB_COMPILE_ERROR (_("Exporting error to file \"%s\""), filename);
90  }
91  cairo_destroy (cairoTarget);
92  cairo_surface_destroy (cSurface);
93 }
