130 static void
131 draw_check_if_object_is_in_selected_area(
132  cairo_t* cairoTarget, gboolean isStroke, gerbv_selection_info_t* selectionInfo, gerbv_image_t* image,
133  struct gerbv_net* net, enum draw_mode drawMode
134 ) {
135  gerbv_selection_item_t sItem = { image, net };
136  gdouble corner1X, corner1Y, corner2X, corner2Y;
137  gdouble x1, x2, y1, y2;
138  gdouble minX, minY, maxX, maxY;
139 
140  corner1X = selectionInfo->lowerLeftX;
141  corner1Y = selectionInfo->lowerLeftY;
142  corner2X = selectionInfo->upperRightX;
143  corner2Y = selectionInfo->upperRightY;
144 
145  /* calculate the coordinate of the user's click in the current
146  transformation matrix */
147  cairo_device_to_user(cairoTarget, &corner1X, &corner1Y);
148  cairo_device_to_user(cairoTarget, &corner2X, &corner2Y);
149 
150  switch (selectionInfo->type) {
151  case GERBV_SELECTION_POINT_CLICK:
152  /* use the cairo in_fill routine to see if the point is within the
153  drawn area */
154  if ((isStroke && cairo_in_stroke(cairoTarget, corner1X, corner1Y))
155  || (!isStroke && cairo_in_fill(cairoTarget, corner1X, corner1Y))) {
156 
157  if (!draw_net_is_in_selection_buffer_remove(net, selectionInfo, (drawMode == FIND_SELECTIONS_TOGGLE))) {
158  selection_add_item(selectionInfo, &sItem);
159  }
160  }
161  break;
162 
163  case GERBV_SELECTION_DRAG_BOX:
164  /* we can't assume the "lowerleft" corner is actually in the lower left,
165  since the cairo transformation matrix may be mirrored,etc */
166  minX = MIN(corner1X, corner2X);
167  maxX = MAX(corner1X, corner2X);
168  minY = MIN(corner1Y, corner2Y);
169  maxY = MAX(corner1Y, corner2Y);
170 
171  if (isStroke)
172  cairo_stroke_extents(cairoTarget, &x1, &y1, &x2, &y2);
173  else
174  cairo_fill_extents(cairoTarget, &x1, &y1, &x2, &y2);
175 
176  if ((minX < x1) && (minY < y1) && (maxX > x2) && (maxY > y2)) {
177  if (!draw_net_is_in_selection_buffer_remove(net, selectionInfo, (drawMode == FIND_SELECTIONS_TOGGLE))) {
178  selection_add_item(selectionInfo, &sItem);
179  }
180  }
181  break;
182  default: break;
183  }
184  /* clear the path, since we didn't actually draw it and cairo
185  doesn't reset it after the previous calls */
186  cairo_new_path(cairoTarget);
187 }

244 static void
245 gerbv_draw_oblong(cairo_t* cairoTarget, gdouble width, gdouble height) {
246  /* --- This stuff produces a line + rounded ends --- */
247  gdouble circleDiameter, strokeDistance;
248 
249  cairo_new_path(cairoTarget);
250  if (width < height) {
251  circleDiameter = width;
252  strokeDistance = (height - width) / 2.0;
253  cairo_arc(cairoTarget, 0.0, strokeDistance, circleDiameter / 2.0, 0, -M_PI);
254  cairo_line_to(cairoTarget, -circleDiameter / 2.0, -strokeDistance);
255  cairo_arc(cairoTarget, 0.0, -strokeDistance, circleDiameter / 2.0, -M_PI, 0);
256  cairo_line_to(cairoTarget, circleDiameter / 2.0, strokeDistance);
257  } else {
258  circleDiameter = height;
259  strokeDistance = (width - height) / 2.0;
260  cairo_arc(cairoTarget, -strokeDistance, 0.0, circleDiameter / 2.0, M_PI_2, -M_PI_2);
261  cairo_line_to(cairoTarget, strokeDistance, -circleDiameter / 2.0);
262  cairo_arc(cairoTarget, strokeDistance, 0.0, circleDiameter / 2.0, -M_PI_2, M_PI_2);
263  cairo_line_to(cairoTarget, -strokeDistance, circleDiameter / 2.0);
264  }
265 
266 #if 0
267  /* --- This stuff produces an oval pad --- */
268  /* cairo doesn't have a function to draw ovals, so we must
269  * draw an arc and stretch it by scaling different x and y values */
270  cairo_save (cairoTarget);
271  cairo_scale (cairoTarget, width, height);
272  gerbv_draw_circle (cairoTarget, 1);
273  cairo_restore (cairoTarget);
274 #endif
275  return;

278 static void
279 gerbv_draw_polygon(cairo_t* cairoTarget, gdouble outsideDiameter, gdouble numberOfSides, gdouble degreesOfRotation) {
280  int i, numberOfSidesInteger = (int)numberOfSides;
281 
282  cairo_rotate(cairoTarget, DEG2RAD(degreesOfRotation));
283  cairo_move_to(cairoTarget, outsideDiameter / 2.0, 0);
284 
285  /* skip first point, since we've moved there already */
286  /* include last point, since we may be drawing an aperture hole next
287  and cairo may not correctly close the path itself */
288  for (i = 1; i <= (int)numberOfSidesInteger; i++) {
289  gdouble angle = ((double)i) * M_PI * 2.0 / numberOfSidesInteger;
290  cairo_line_to(cairoTarget, cos(angle) * outsideDiameter / 2.0, sin(angle) * outsideDiameter / 2.0);
291  }
292 
293  return;
294 }

593  draw_fill(cairoTarget, drawMode, selectionInfo, image, net);
594 
595  cairo_restore(cairoTarget);
596 
597  break;
598  }
599 
600  draw_fill(cairoTarget, drawMode, selectionInfo, image, net);
601  break;
602 
603  default: GERB_COMPILE_WARNING(_("Unknown macro type: %s"), gerbv_aperture_type_name(ls->type)); ret = 0;
604  }
605 
606  cairo_restore(cairoTarget);
607  ls = ls->next;
608  }
609 
610  if (usesClearPrimitive) {
611  cairo_pop_group_to_source(cairoTarget);
612  cairo_paint(cairoTarget);
613  }
614 
615  return ret;
616 }

618 void
619 draw_apply_netstate_transformation(cairo_t* cairoTarget, gerbv_netstate_t* state) {
620  /* apply scale factor */
621  cairo_scale(cairoTarget, state->scaleA, state->scaleB);
622  /* apply offset */
623  cairo_translate(cairoTarget, state->offsetA, state->offsetB);
624  /* apply mirror */
625  switch (state->mirrorState) {
626  case GERBV_MIRROR_STATE_FLIPA: cairo_scale(cairoTarget, -1, 1); break;
627  case GERBV_MIRROR_STATE_FLIPB: cairo_scale(cairoTarget, 1, -1); break;
628  case GERBV_MIRROR_STATE_FLIPAB: cairo_scale(cairoTarget, -1, -1); break;
629  default: break;
630  }
631  /* finally, apply axis select */
632  if (state->axisSelect == GERBV_AXIS_SELECT_SWAPAB) {
633  /* we do this by rotating 270 (counterclockwise, then mirroring
634  the Y axis */
635  cairo_rotate(cairoTarget, M_PI + M_PI_2);
636  cairo_scale(cairoTarget, 1, -1);
637  }
638 }

640 void
641 draw_render_polygon_object(
642  gerbv_net_t* oldNet, cairo_t* cairoTarget, gdouble sr_x, gdouble sr_y, gerbv_image_t* image,
643  enum draw_mode drawMode, gerbv_selection_info_t* selectionInfo, gboolean pixelOutput
644 ) {
645  gerbv_net_t *currentNet, *polygonStartNet;
646  int haveDrawnFirstFillPoint = 0;
647  gdouble x2, y2, cp_x = 0, cp_y = 0;
648 
649  haveDrawnFirstFillPoint = FALSE;
650  /* save the first net in the polygon as the "ID" net pointer
651  in case we are saving this net to the selection array */
652  polygonStartNet = oldNet;
653  cairo_new_path(cairoTarget);
654 
655  for (currentNet = oldNet->next; currentNet != NULL; currentNet = currentNet->next) {
656  x2 = currentNet->stop_x + sr_x;
657  y2 = currentNet->stop_y + sr_y;
658 
659  /* translate circular x,y data as well */
660  if (currentNet->cirseg) {
661  cp_x = currentNet->cirseg->cp_x + sr_x;
662  cp_y = currentNet->cirseg->cp_y + sr_y;
663  }
664  if (!haveDrawnFirstFillPoint) {
665  draw_cairo_move_to(cairoTarget, x2, y2, FALSE, pixelOutput);
666  haveDrawnFirstFillPoint = TRUE;
667  continue;
668  }
669 
670  switch (currentNet->interpolation) {
671  case GERBV_INTERPOLATION_LINEARx1:
672  case GERBV_INTERPOLATION_LINEARx10:
673  case GERBV_INTERPOLATION_LINEARx01:
674  case GERBV_INTERPOLATION_LINEARx001: draw_cairo_line_to(cairoTarget, x2, y2, FALSE, pixelOutput); break;
675  case GERBV_INTERPOLATION_CW_CIRCULAR:
676  case GERBV_INTERPOLATION_CCW_CIRCULAR:
677  if (currentNet->cirseg->angle2 > currentNet->cirseg->angle1) {
678  cairo_arc(
679  cairoTarget, cp_x, cp_y, currentNet->cirseg->width / 2.0, DEG2RAD(currentNet->cirseg->angle1),
680  DEG2RAD(currentNet->cirseg->angle2)
681  );
682  } else {
683  cairo_arc_negative(
684  cairoTarget, cp_x, cp_y, currentNet->cirseg->width / 2.0, DEG2RAD(currentNet->cirseg->angle1),
685  DEG2RAD(currentNet->cirseg->angle2)
686  );
687  }
688  break;
689  case GERBV_INTERPOLATION_PAREA_END:
690  cairo_close_path(cairoTarget);
691  /* turn off anti-aliasing for polygons, since it shows seams
692  with adjacent polygons (usually on PCB ground planes) */
693  cairo_antialias_t oldAlias = cairo_get_antialias(cairoTarget);
694  cairo_set_antialias(cairoTarget, CAIRO_ANTIALIAS_NONE);
695  draw_fill(cairoTarget, drawMode, selectionInfo, image, polygonStartNet);
696  cairo_set_antialias(cairoTarget, oldAlias);
697  return;
698  default: break;
699  }
700  }

717 static int
718 draw_calc_pnp_mark_coords(struct gerbv_net* start_net, double* label_x, double* label_y) {
719  double x, y;
720  struct gerbv_net* net = start_net;
721  const char* label = NULL;
722 
723  if (net && net->label)
724  label = net->label->str;
725 
726  if (label == NULL)
727  return 0;
728 
729  x = HUGE_VAL;
730  y = -HUGE_VAL;
731  do {
732  if (!net->label || 0 != g_strcmp0(net->label->str, label))
733  break;
734 
735  /* Search top left corner */
736  if (net->boundingBox.top != HUGE_VAL) {
737  /* Bounding box not calculated */
738  x = MIN(x, net->boundingBox.left);
739  y = MAX(y, net->boundingBox.top);
740  } else {
741  x = MIN(x, net->stop_x);
742  y = MAX(y, net->stop_y + 0.01 / 2);
743  /* 0.01 default line width */
744  }
745  } while (NULL != (net = gerbv_image_return_next_renderable_object(net)));
746 
747  *label_x = x;
748  *label_y = y;
749 
750  return 1;
751 }

753 int
754 draw_image_to_cairo_target(
755  cairo_t* cairoTarget, gerbv_image_t* image, gdouble pixelWidth, enum draw_mode drawMode,
756  gerbv_selection_info_t* selectionInfo, gerbv_render_info_t* renderInfo, gboolean allowOptimization,
757  gerbv_user_transformation_t transform, gboolean pixelOutput
758 ) {
759  const int hole_cross_inc_px = 8;
760  struct gerbv_net *net, *polygonStartNet = NULL;
761  double x1, y1, x2, y2, cp_x = 0, cp_y = 0;
762  gdouble * p, p0, p1, dx, dy, lineWidth, r;
763  gerbv_netstate_t* oldState;
764  gerbv_layer_t* oldLayer;
765  cairo_operator_t drawOperatorClear, drawOperatorDark;
766  gboolean invertPolarity = FALSE, oddWidth = FALSE;
767  gdouble minX = 0, minY = 0, maxX = 0, maxY = 0;
768  gdouble criticalRadius;
769  gdouble scaleX = transform.scaleX;
770  gdouble scaleY = transform.scaleY;
771  /* Keep PNP label not mirrored */
772  gdouble pnp_label_scale_x = 1, pnp_label_scale_y = -1;
773  gboolean limitLineWidth = TRUE;
774  gboolean displayPixel = TRUE;
775  gboolean doVectorExportFix;
776  double bg_r, bg_g, bg_b; /* Background color */
777 
778  doVectorExportFix = draw_do_vector_export_fix(cairoTarget, &bg_r, &bg_g, &bg_b);
779 
780  /* If we are scaling the image at all, ignore the line width checks
781  * since scaled up lines can still be visible */
782  if ((scaleX != 1) || (scaleY != 1)) {
783  limitLineWidth = FALSE;
784  }
785 
786  if (transform.mirrorAroundX) {
787  scaleY *= -1;
788  pnp_label_scale_y = 1;
789  }
790 
791  if (transform.mirrorAroundY) {
792  scaleX *= -1;
793  pnp_label_scale_x = -1;
794  }
795 
796  cairo_translate(cairoTarget, transform.translateX, transform.translateY);
797  cairo_scale(cairoTarget, scaleX, scaleY);
798  cairo_rotate(cairoTarget, transform.rotation);
799 
800  gboolean useOptimizations = allowOptimization;
801 
802  /* If the user is using any transformations for this layer, then don't
803  * bother using rendering optimizations */
804  if (fabs(transform.translateX) > GERBV_PRECISION_LINEAR_INCH
805  || fabs(transform.translateY) > GERBV_PRECISION_LINEAR_INCH
806  || fabs(transform.scaleX - 1) > GERBV_PRECISION_LINEAR_INCH
807  || fabs(transform.scaleY - 1) > GERBV_PRECISION_LINEAR_INCH
808  || fabs(transform.rotation) > GERBV_PRECISION_ANGLE_RAD || transform.mirrorAroundX || transform.mirrorAroundY)
809  useOptimizations = FALSE;
810 
811  if (useOptimizations && pixelOutput) {
812  minX = renderInfo->lowerLeftX;
813  minY = renderInfo->lowerLeftY;
814  maxX = renderInfo->lowerLeftX + (renderInfo->displayWidth / renderInfo->scaleFactorX);
815  maxY = renderInfo->lowerLeftY + (renderInfo->displayHeight / renderInfo->scaleFactorY);
816  }
817 
818  /* do initial justify */
819  cairo_translate(cairoTarget, image->info->imageJustifyOffsetActualA, image->info->imageJustifyOffsetActualB);
820 
821  /* set the fill rule so aperture holes are cleared correctly */
822  cairo_set_fill_rule(cairoTarget, CAIRO_FILL_RULE_EVEN_ODD);
823  /* offset image */
824  cairo_translate(cairoTarget, image->info->offsetA, image->info->offsetB);
825  /* do image rotation */
826  cairo_rotate(cairoTarget, image->info->imageRotation);
827 
828  /* load in polarity operators depending on the image polarity */
829  invertPolarity = transform.inverted;
830  if (image->info->polarity == GERBV_POLARITY_NEGATIVE)
831  invertPolarity = !invertPolarity;
832  if (drawMode == DRAW_SELECTIONS)
833  invertPolarity = FALSE;
834 
835  if (invertPolarity) {
836  drawOperatorClear = CAIRO_OPERATOR_OVER;
837  drawOperatorDark = CAIRO_OPERATOR_CLEAR;
838  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_OVER);
839  cairo_paint(cairoTarget);
840  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_CLEAR);
841  } else {
842  drawOperatorClear = CAIRO_OPERATOR_CLEAR;
843  drawOperatorDark = CAIRO_OPERATOR_OVER;
844  }
845 
846  /* next, push two cairo states to simulate the first layer and netstate
847  translations (these will be popped when another layer or netstate is
848  started */
849  cairo_save(cairoTarget);
850  cairo_save(cairoTarget);
851 
852  /* store the current layer and netstate so we know when they change */
853  oldLayer = image->layers;
854  oldState = image->states;
855 
856  const char* pnp_net_label_str_prev = NULL;
857 
858  for (net = image->netlist->next; net != NULL; net = gerbv_image_return_next_renderable_object(net)) {
859 
860  /* check if this is a new layer */
861  if (net->layer != oldLayer) {
862  /* it's a new layer, so recalculate the new transformation matrix
863  for it */
864  cairo_restore(cairoTarget);
865  cairo_restore(cairoTarget);
866  cairo_save(cairoTarget);
867  /* do any rotations */
868  cairo_rotate(cairoTarget, net->layer->rotation);
869  /* handle the layer polarity */
870  if ((net->layer->polarity == GERBV_POLARITY_CLEAR) ^ invertPolarity) {
871  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_CLEAR);
872  drawOperatorClear = CAIRO_OPERATOR_OVER;
873  drawOperatorDark = CAIRO_OPERATOR_CLEAR;
874  } else {
875  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_OVER);
876  drawOperatorClear = CAIRO_OPERATOR_CLEAR;
877  drawOperatorDark = CAIRO_OPERATOR_OVER;
878  }
879 
880  /* Draw any knockout areas */
881  gerbv_knockout_t* ko = &net->layer->knockout;
882  if (ko->firstInstance == TRUE) {
883  cairo_save(cairoTarget);
884 
885  if (ko->polarity == GERBV_POLARITY_CLEAR) {
886  cairo_set_operator(cairoTarget, drawOperatorClear);
887  } else {
888  cairo_set_operator(cairoTarget, drawOperatorDark);
889  }
890 
891  if (doVectorExportFix && CAIRO_OPERATOR_CLEAR == cairo_get_operator(cairoTarget)) {
892 
893  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_OVER);
894  cairo_set_source_rgba(cairoTarget, bg_r, bg_g, bg_b, 1.0);
895  }
896 
897  cairo_new_path(cairoTarget);
898  cairo_rectangle(
899  cairoTarget, ko->lowerLeftX - ko->border, ko->lowerLeftY - ko->border, ko->width + 2 * ko->border,
900  ko->height + 2 * ko->border
901  );
902  draw_fill(cairoTarget, drawMode, selectionInfo, image, net);
903 
904  cairo_restore(cairoTarget);
905  }
906 
907  /* Finally, reapply old netstate transformation */
908  cairo_save(cairoTarget);
909  draw_apply_netstate_transformation(cairoTarget, net->state);
910  oldLayer = net->layer;
911  }
912 
913  /* check if this is a new netstate */
914  if (net->state != oldState) {
915  /* pop the transformation matrix back to the "pre-state" state and
916  resave it */
917  cairo_restore(cairoTarget);
918  cairo_save(cairoTarget);
919  /* it's a new state, so recalculate the new transformation matrix
920  for it */
921  draw_apply_netstate_transformation(cairoTarget, net->state);
922  oldState = net->state;
923  }
924 
925  /* if we are only drawing from the selection buffer, search if this net is
926  in the buffer */
927  if (drawMode == DRAW_SELECTIONS) {
928  /* this flag makes sure we don't draw any unintentional polygons...
929  if we've successfully entered a polygon (the first net matches, and
930  we don't want to check the nets inside the polygon) then
931  polygonStartNet will be set */
932  if (!polygonStartNet) {
933  if (!draw_net_is_in_selection_buffer_remove(net, selectionInfo, FALSE))
934  continue;
935  }
936  }
937 
938  /* Render any labels attached to this net */
939  /* NOTE: this is currently only used on PNP files, so we may
940  make some assumptions here... */
941  if (drawMode != DRAW_SELECTIONS && net->label
942  && (image->layertype == GERBV_LAYERTYPE_PICKANDPLACE_TOP
943  || image->layertype == GERBV_LAYERTYPE_PICKANDPLACE_BOT)
944  && g_strcmp0(net->label->str, pnp_net_label_str_prev)) {
945 
946  double mark_x, mark_y;
947 
948  /* Add PNP text label only one time per
949  * net and if it is not selected. */
950  pnp_net_label_str_prev = net->label->str;
951 
952  if (draw_calc_pnp_mark_coords(net, &mark_x, &mark_y)) {
953  cairo_save(cairoTarget);
954 
955  cairo_set_font_size(cairoTarget, 0.05);
956  cairo_move_to(cairoTarget, mark_x, mark_y);
957  cairo_scale(cairoTarget, pnp_label_scale_x, pnp_label_scale_y);
958  cairo_show_text(cairoTarget, net->label->str);
959 
960  cairo_restore(cairoTarget);
961  }
962  }
963 
964  /* step and repeat */
965  gerbv_step_and_repeat_t* sr = &net->layer->stepAndRepeat;
966  int ix, iy;
967  for (ix = 0; ix < sr->X; ix++) {
968  for (iy = 0; iy < sr->Y; iy++) {
969  double sr_x = ix * sr->dist_X;
970  double sr_y = iy * sr->dist_Y;
971 
972  if (useOptimizations && pixelOutput
973  && ((net->boundingBox.right + sr_x < minX) || (net->boundingBox.left + sr_x > maxX)
974  || (net->boundingBox.top + sr_y < minY) || (net->boundingBox.bottom + sr_y > maxY))) {
975  continue;
976  }
977 
978  x1 = net->start_x + sr_x;
979  y1 = net->start_y + sr_y;
980  x2 = net->stop_x + sr_x;
981  y2 = net->stop_y + sr_y;
982 
983  /* translate circular x,y data as well */
984  if (net->cirseg) {
985  cp_x = net->cirseg->cp_x + sr_x;
986  cp_y = net->cirseg->cp_y + sr_y;
987  }
988 
989  /* Polygon area fill routines */
990  switch (net->interpolation) {
991  case GERBV_INTERPOLATION_PAREA_START:
992 
993  if (doVectorExportFix && CAIRO_OPERATOR_CLEAR == cairo_get_operator(cairoTarget)) {
994 
995  cairo_save(cairoTarget);
996 
997  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_OVER);
998  cairo_set_source_rgba(cairoTarget, bg_r, bg_g, bg_b, 1.0);
999 
1000  draw_render_polygon_object(
1001  net, cairoTarget, sr_x, sr_y, image, drawMode, selectionInfo, pixelOutput
1002  );
1003 
1004  cairo_restore(cairoTarget);
1005  } else {
1006  draw_render_polygon_object(
1007  net, cairoTarget, sr_x, sr_y, image, drawMode, selectionInfo, pixelOutput
1008  );
1009  }
1010 
1011  continue;
1012  case GERBV_INTERPOLATION_DELETED: continue;
1013  default: break;
1014  }
1015 
1016  /*
1017  * If aperture state is off we allow use of undefined apertures.
1018  * This happens when gerber files starts, but hasn't decided on
1019  * which aperture to use.
1020  */
1021  if (image->aperture[net->aperture] == NULL)
1022  continue;
1023 
1024  switch (net->aperture_state) {
1025  case GERBV_APERTURE_STATE_ON:
1026  /* if the aperture width is truly 0, then render as a 1 pixel width
1027  line. 0 diameter apertures are used by some programs to draw labels,
1028  etc, and they are rendered by other programs as 1 pixel wide */
1029  /* NOTE: also, make sure all lines are at least 1 pixel wide, so they
1030  always show up at low zoom levels */
1031 
1032  if (limitLineWidth
1033  && ((image->aperture[net->aperture]->parameter[0] < pixelWidth) && (pixelOutput)))
1034  criticalRadius = pixelWidth / 2.0;
1035  else
1036  criticalRadius = image->aperture[net->aperture]->parameter[0] / 2.0;
1037  lineWidth = criticalRadius * 2.0;
1038  // convert to a pixel integer
1039  cairo_user_to_device_distance(cairoTarget, &lineWidth, &x1);
1040  if (pixelOutput) {
1041  lineWidth = round(lineWidth);
1042  if ((int)lineWidth % 2) {
1043  oddWidth = TRUE;
1044  } else {
1045  oddWidth = FALSE;
1046  }
1047  }
1048  cairo_device_to_user_distance(cairoTarget, &lineWidth, &x1);
1049  cairo_set_line_width(cairoTarget, lineWidth);
1050 
1051  switch (net->interpolation) {
1052  case GERBV_INTERPOLATION_LINEARx1:
1053  case GERBV_INTERPOLATION_LINEARx10:
1054  case GERBV_INTERPOLATION_LINEARx01:
1055  case GERBV_INTERPOLATION_LINEARx001:
1056  cairo_set_line_cap(cairoTarget, CAIRO_LINE_CAP_ROUND);
1057 
1058  /* weed out any lines that are
1059  * obviously not going to
1060  * render on the visible screen */
1061  switch (image->aperture[net->aperture]->type) {
1062  case GERBV_APTYPE_CIRCLE:
1063  if (renderInfo->show_cross_on_drill_holes
1064  && image->layertype == GERBV_LAYERTYPE_DRILL) {
1065  /* Draw center crosses on slot hole */
1066  cairo_set_line_width(cairoTarget, pixelWidth);
1067  cairo_set_line_cap(cairoTarget, CAIRO_LINE_CAP_SQUARE);
1068  r = image->aperture[net->aperture]->parameter[0] / 2.0
1069  + hole_cross_inc_px * pixelWidth;
1070  draw_cairo_cross(cairoTarget, x1, y1, r);
1071  draw_cairo_cross(cairoTarget, x2, y2, r);
1072  cairo_set_line_cap(cairoTarget, CAIRO_LINE_CAP_ROUND);
1073  cairo_set_line_width(cairoTarget, lineWidth);
1074  }
1075 
1076  draw_cairo_move_to(cairoTarget, x1, y1, oddWidth, pixelOutput);
1077  draw_cairo_line_to(cairoTarget, x2, y2, oddWidth, pixelOutput);
1078 
1079  if (doVectorExportFix
1080  && CAIRO_OPERATOR_CLEAR == cairo_get_operator(cairoTarget)) {
1081  cairo_save(cairoTarget);
1082  cairo_set_source_rgba(cairoTarget, bg_r, bg_g, bg_b, 1.0);
1083  cairo_set_operator(cairoTarget, CAIRO_OPERATOR_OVER);
1084 
1085  draw_stroke(cairoTarget, drawMode, selectionInfo, image, net);
1086 
1087  cairo_restore(cairoTarget);
1088  } else {
1089  draw_stroke(cairoTarget, drawMode, selectionInfo, image, net);
1090  }
1091 
1092  break;
1093  case GERBV_APTYPE_RECTANGLE:
1094  dx = image->aperture[net->aperture]->parameter[0] / 2;
1095  dy = image->aperture[net->aperture]->parameter[1] / 2;
1096  if (x1 > x2)
1097  dx = -dx;
1098  if (y1 > y2)
1099  dy = -dy;
1100  cairo_new_path(cairoTarget);
1101  draw_cairo_move_to(cairoTarget, x1 - dx, y1 - dy, FALSE, pixelOutput);
1102  draw_cairo_line_to(cairoTarget, x1 - dx, y1 + dy, FALSE, pixelOutput);
1103  draw_cairo_line_to(cairoTarget, x2 - dx, y2 + dy, FALSE, pixelOutput);
1104  draw_cairo_line_to(cairoTarget, x2 + dx, y2 + dy, FALSE, pixelOutput);
1105  draw_cairo_line_to(cairoTarget, x2 + dx, y2 - dy, FALSE, pixelOutput);
1106  draw_cairo_line_to(cairoTarget, x1 + dx, y1 - dy, FALSE, pixelOutput);
1107  draw_fill(cairoTarget, drawMode, selectionInfo, image, net);
1108  break;
1109  /* TODO: for now, just render ovals or polygons like a circle */
1110  case GERBV_APTYPE_OVAL:
1111  case GERBV_APTYPE_POLYGON:
1112  draw_cairo_move_to(cairoTarget, x1, y1, oddWidth, pixelOutput);
1113  draw_cairo_line_to(cairoTarget, x2, y2, oddWidth, pixelOutput);
1114  draw_stroke(cairoTarget, drawMode, selectionInfo, image, net);
1115  break;
1116  /* macros can only be flashed, so ignore any that might be here */
1117  default:
1118  GERB_COMPILE_WARNING(
1119  _("Unknown aperture type: %s"),
1120  _(gerbv_aperture_type_name(image->aperture[net->aperture]->type))
1121  );
1122  break;
1123  }
1124  break;
1125  case GERBV_INTERPOLATION_CW_CIRCULAR:
1126  case GERBV_INTERPOLATION_CCW_CIRCULAR:
1127  /* cairo doesn't have a function to draw oval arcs, so we must
1128  * draw an arc and stretch it by scaling different x and y values
1129  */
1130  cairo_new_path(cairoTarget);
1131  if (image->aperture[net->aperture]->type == GERBV_APTYPE_RECTANGLE) {
1132  cairo_set_line_cap(cairoTarget, CAIRO_LINE_CAP_SQUARE);
1133  } else {
1134  cairo_set_line_cap(cairoTarget, CAIRO_LINE_CAP_ROUND);
1135  }
1136  cairo_save(cairoTarget);
